## API Reference

### Autocomplete (Root)

<PropsTable
  props={[
    { name: "items", type: "T[] | GroupedItems[]", description: "The items to display. Can be a flat array or an array of groups (objects with value and items properties)." },
    { name: "mode", type: "'list' | 'both' | 'inline' | 'none'", default: "'list'", description: "Controls filtering and inline autocompletion. 'list' filters items, 'both' filters and shows inline completion, 'inline' shows inline completion without filtering, 'none' disables both." },
    { name: "openOnInputClick", type: "boolean", default: "false", description: "Whether the popup opens when clicking the input." },
    { name: "value", type: "string", description: "The controlled input value." },
    { name: "defaultValue", type: "string", description: "The uncontrolled initial input value." },
    { name: "onValueChange", type: "(value: string, eventDetails) => void", description: "Callback fired when the input value changes." },
    { name: "onOpenChange", type: "(open: boolean, eventDetails) => void", description: "Callback fired when the popup is opened or closed." },
    { name: "autoHighlight", type: "boolean | 'always'", default: "false", description: "Whether the first matching item is highlighted automatically." },
    { name: "keepHighlight", type: "boolean", default: "false", description: "Whether the highlighted item is preserved when the pointer leaves the list." },
    { name: "disabled", type: "boolean", default: "false", description: "Whether the component should ignore user interaction." }
  ]}
/>
