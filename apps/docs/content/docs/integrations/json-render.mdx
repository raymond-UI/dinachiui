## Overview

`@dinachi/json-render` is a first-party adapter for [json-render](https://json-render.dev/) (Vercel Labs' generative UI framework). It lets AI models generate JSON specs that render as fully accessible DinachiUI components.

The package provides:

- **Catalog** — Zod-validated schemas for 30 DinachiUI components + 3 actions (server-safe, no React dependency)
- **Registry** — React implementations mapping JSON props to real DinachiUI components
- **Factories** — Helper functions for subsetting and customizing the catalog/registry

## Installation

```bash
npm install @dinachi/json-render @json-render/core @json-render/react zod
```

## Quick Start

```tsx
import {
  createDinachiCatalog,
  createDinachiRegistry,
  JSONUIProvider,
  Renderer,
} from "@dinachi/json-render";
import type { Spec } from "@dinachi/json-render";

// 1. Create catalog + registry
const catalog = createDinachiCatalog();
const { registry } = createDinachiRegistry(catalog);

// 2. Define a UI spec (typically generated by an AI model)
const spec: Spec = {
  root: "card",
  elements: {
    card: {
      type: "Card",
      props: { title: "Welcome", description: "Get started with DinachiUI" },
      children: ["greeting", "actionBtn"],
    },
    greeting: {
      type: "Badge",
      props: { text: "New", variant: "success" },
    },
    actionBtn: {
      type: "Button",
      props: { label: "Get Started" },
      on: { press: { action: "navigate", params: { url: "/docs" } } },
    },
  },
};

// 3. Render
function App() {
  return (
    <JSONUIProvider registry={registry} initialState={{}}>
      <Renderer spec={spec} registry={registry} />
    </JSONUIProvider>
  );
}
```

<Callout type="warning" title="Important">
Always import `Renderer`, `JSONUIProvider`, and other React providers from `@dinachi/json-render` — not directly from `@json-render/react`. Direct imports can cause dual-package context mismatches when zod versions differ.
</Callout>

## Catalog API

### Full catalog

```ts
import { catalog, dinachiComponentDefinitions, dinachiActionDefinitions } from "@dinachi/json-render";
```

The default `catalog` includes all 22 components and 3 actions. Use `dinachiComponentDefinitions` and `dinachiActionDefinitions` to inspect available schemas.

For server-side usage (API routes), import from the catalog subpath to avoid pulling in React:

```ts
import { dinachiComponentDefinitions, dinachiActionDefinitions } from "@dinachi/json-render/catalog";
```

### Custom catalog with `createDinachiCatalog()`

Subset components for specific use cases:

```ts
import { createDinachiCatalog } from "@dinachi/json-render";

// Form-only catalog
const formCatalog = createDinachiCatalog({
  components: ["Input", "Textarea", "Select", "Checkbox", "Button", "Card", "Box", "Text"],
});

// Exclude actions
const noActionsCatalog = createDinachiCatalog({
  includeActions: false,
});

// Extend with custom components
const extendedCatalog = createDinachiCatalog({
  extend: {
    CustomWidget: {
      props: z.object({ title: z.string() }),
      description: "A custom widget",
    },
  },
});
```

## Registry API

### `createDinachiRegistry(catalog, options?)`

Creates a component registry from a catalog. Returns `{ registry, handlers, executeAction }`.

```ts
import { createDinachiCatalog, createDinachiRegistry } from "@dinachi/json-render";

const catalog = createDinachiCatalog();
const { registry, handlers, executeAction } = createDinachiRegistry(catalog);
```

Override specific components or actions:

```ts
const { registry } = createDinachiRegistry(catalog, {
  componentOverrides: {
    Button: ({ props, emit }) => (
      <button className="custom-btn" onClick={() => emit?.("press")}>
        {props.label}
      </button>
    ),
  },
  actionOverrides: {
    navigate: async (params) => {
      router.push(params.url); // Use your app's router
    },
  },
});
```

## Actions

Actions are triggered by event bindings in the spec's `on` field.

| Action | Params | Description |
|--------|--------|-------------|
| **navigate** | `url`, `target?` | Navigate to a URL. Set `target: "_blank"` for new tab. |
| **submit** | `formId?` | Submit a form element by ID. |
| **showToast** | `title`, `description?`, `variant?`, `timeout?` | Show a toast notification. |

Example event binding:

```json
{
  "type": "Button",
  "props": { "label": "Save" },
  "on": {
    "press": {
      "action": "showToast",
      "params": { "title": "Saved!", "variant": "success" }
    }
  }
}
```

## State Binding

Input components support two-way state binding via the `statePath` prop. State paths use JSON Pointer syntax (e.g., `/form/email`).

```json
{
  "root": "form",
  "state": {
    "form": { "email": "", "agreed": false }
  },
  "elements": {
    "form": {
      "type": "Card",
      "props": { "title": "Sign Up" },
      "children": ["emailInput", "agreeCheck"]
    },
    "emailInput": {
      "type": "Input",
      "props": {
        "label": "Email",
        "type": "email",
        "statePath": "/form/email"
      }
    },
    "agreeCheck": {
      "type": "Checkbox",
      "props": {
        "label": "I agree to terms",
        "statePath": "/form/agreed"
      }
    }
  }
}
```

## Full App Example

```tsx
"use client";

import {
  createDinachiCatalog,
  createDinachiRegistry,
  JSONUIProvider,
  Renderer,
  toastManager,
} from "@dinachi/json-render";
import type { Spec } from "@dinachi/json-render";
import { Toast } from "@/components/ui/toast";
import { useMemo } from "react";

export function GenerativeUI({ spec }: { spec: Spec }) {
  const { registry } = useMemo(() => {
    const catalog = createDinachiCatalog();
    return createDinachiRegistry(catalog);
  }, []);

  return (
    <JSONUIProvider
      registry={registry}
      initialState={spec.state ?? {}}
    >
      <Renderer spec={spec} registry={registry} />
      <Toast toastManager={toastManager} />
    </JSONUIProvider>
  );
}
```
